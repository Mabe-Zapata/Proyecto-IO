    9, 12, 13, 7,
    14,  9, 16, 5
    ), nrow = 3, byrow = TRUE)

    # Capacidades (oferta)
    capacidad <- c(50, 60, 40)

    # Demandas
    demanda <- c(30, 45, 35, 40)

    # Resolver modelo de transporte
    transporte <- lp.transport(
    cost.mat  = costos,
    direction = "min",
    row.signs = rep("<=", 3),
    row.rhs   = capacidad,
    col.signs = rep(">=", 4),
    col.rhs   = demanda
    )
